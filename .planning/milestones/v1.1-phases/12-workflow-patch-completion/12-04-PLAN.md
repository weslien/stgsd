- stclaude get-state --json

- stclaude get-phase <N> --json
  Returns: { phase{number,name,slug,goal,status}, plans[], requirements[] }

- stclaude roadmap analyze --json
  Returns: { phases[{number,name,status,goal}], milestones[], requirements[] }
- stclaude add-phase --number <N> --name <name> --goal <goal> --json
  Adds new phase to SpacetimeDB
- stclaude init execute-phase <phase> --json
  Returns plans with hasSummary flags
- stclaude init plan-phase <phase> --json
  Returns phase details and existing context/research
