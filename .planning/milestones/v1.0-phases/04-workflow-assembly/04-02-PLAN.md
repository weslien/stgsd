**Helper pattern for insert confirmation (define locally in each file):**

All write commands need a `waitForInsert` pattern. Define it locally in each file since the match function varies per table:

```typescript
function waitForInsert(
  conn: DbConnection,
  table: { onInsert: (cb: (_ctx: any, row: any) => void) => void },
  matchFn: (row: any) => boolean,
  timeoutMs = 5000,
): Promise<void> {
  return new Promise((resolve, reject) => {
    const timer = setTimeout(() => {
      reject(new CliError(ErrorCodes.INTERNAL_ERROR, 'Insert confirmation timed out after 5 seconds'));
    }, timeoutMs);

    table.onInsert((_ctx: any, row: any) => {
      if (matchFn(row)) {
        clearTimeout(timer);
        resolve();
      }
    });
  });
}
```
