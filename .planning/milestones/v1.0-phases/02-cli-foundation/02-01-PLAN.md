```typescript
import { execSync } from 'node:child_process';
import { CliError, ErrorCodes } from './errors.js';

export function getGitRemoteUrl(): string {
  try {
    return execSync('git config --get remote.origin.url', {
      encoding: 'utf-8',
      timeout: 5000,
    }).trim();
  } catch {
    throw new CliError(
      ErrorCodes.NOT_GIT_REPO,
      'Not in a git repository with a remote. Run from a git repo with an origin remote.',
    );
  }
}
```

Success envelope (per locked decision -- `{ ok: true, data }`):
```typescript
export function outputSuccess(data: unknown, opts: { json: boolean }, humanFormatter?: (data: unknown) => string): void {
  if (opts.json) {
    console.log(JSON.stringify({ ok: true, data }, bigintReplacer));
  } else if (humanFormatter) {
    console.log(humanFormatter(data));
  } else {
    console.log(JSON.stringify(data, bigintReplacer, 2));
  }
}
```

Error envelope (per locked decision -- `{ ok: false, error: { code, message } }`):
```typescript
export function outputError(code: string, message: string, opts: { json: boolean }): void {
  if (opts.json) {
    console.log(JSON.stringify({ ok: false, error: { code, message } }));
  } else {
    console.error(`Error: ${message}`);
  }
  process.exit(1);
}
```

**Create `spacetimeclaude/src/cli/lib/connection.ts`:**

The connect-per-command pattern. Connect to maincloud, subscribe to all tables, run the callback once data is loaded, disconnect and exit.

```typescript
import { DbConnection } from '../../module_bindings/index.js';
import { CliError, ErrorCodes } from './errors.js';

const SPACETIMEDB_URI = 'https://maincloud.spacetimedb.com';
const DATABASE_NAME = 'spacetimeclaude-gvhsi';

export async function withConnection<T>(
  fn: (conn: DbConnection) => T | Promise<T>,
): Promise<T> {
  return new Promise<T>((resolve, reject) => {
    const timeout = setTimeout(() => {
      reject(new CliError(ErrorCodes.CONNECTION_FAILED, 'Connection timed out after 15 seconds'));
    }, 15_000);

    const conn = DbConnection.builder()
      .withUri(SPACETIMEDB_URI)
      .withDatabaseName(DATABASE_NAME)
      .onConnect((conn) => {
        conn.subscriptionBuilder()
          .onApplied(async () => {
            try {
              const result = await fn(conn);
              clearTimeout(timeout);
              conn.disconnect();
              resolve(result);
            } catch (err) {
              clearTimeout(timeout);
              conn.disconnect();
              reject(err);
            }
          })
          .onError((_ctx, err) => {
            clearTimeout(timeout);
            conn.disconnect();
            reject(new CliError(ErrorCodes.CONNECTION_FAILED, `Subscription error: ${err}`));
          })
          .subscribeToAll();
      })
      .onConnectError((_ctx, err) => {
        clearTimeout(timeout);
        reject(new CliError(ErrorCodes.CONNECTION_FAILED, `Connection failed: ${err}`));
      })
      .build();
  });
}
```

Key design decisions:
- `subscribeToAll()` for simplicity (data volume is small -- single project, ~20 rows)
- 15-second timeout prevents hanging if maincloud is unreachable
- Explicit `process.exit()` is NOT in this helper -- that happens in the output layer or the caller. The connection helper just resolves/rejects.
